WHILE $LAUNDRY_BASKET_NOT_EMPTY
REMOVE ITEM FROM LAUNDRY BASKET
IF $CLOTHES = "inside out"
TURN CLOTHES RIGHT SIDE OUT
END IF
IF $CLOTHES_TYPE = "bra"
IF $BRA_PILE_EXISTS
PLACE BRA IN BRA PILE
ELSE
CREATE NEW BRA PILE
PLACE BRA IN BRA PILE
END IF
ELSE IF $CLOTHES_TYPE = "underwear"
IF $UNDERWEAR_PILE_EXISTS
PLACE UNDERWEAR IN UNDERWEAR PILE
ELSE
CREATE NEW UNDERWEAR PILE
PLACE UNDERWEAR IN UNDERWEAR PILE
END IF
ELSE IF $CLOTHES_TYPE = "jeans"
IF $JEANS PILE EXISTS
PLACE JEANS IN JEANS PILE
ELSE
CREATE NEW JEANS PILE
PLACE JEANS IN JEANS PILE
END IF
ELSE IF $CLOTHES_TYPE = "sweatshirt"
IF $SWEATSHIRT PILE EXISTS
PLACE SWEATSHIRT IN SWEATSHIRT PILE
ELSE
CREATE NEW SWEATSHIRT PILE
PLACE_SWEATSHIRT IN SWEATSHIRT PILE
END IF
ELSE IF $CLOTHES_TYPE = "leggings"
IF $LEGGINGS PILE EXISTS
PLACE LEGGINGS IN LEGGINGS PILE
ELSE
CREATE NEW LEGGINGS PILE
PLACE LEGGINGS IN LEGGINGS PILE
END IF
ELSE IF $CLOTHES_TYPE = "workout shirt"
IF $WORKOUT SHIRT PILE EXISTS
PLACE WORKOUT SHIRT IN WORKOUT SHIRT PILE
ELSE
CREATE NEW WORKOUT SHIRT PILE
PLACE WORKOUT SHIRT IN WORKOUT SHIRT PILE
END IF
ELSE IF $CLOTHES_TYPE = "shorts"
IF $SHORTS PILE EXISTS
PLACE SHORTS IN SHORTS PILE
ELSE
CREATE NEW SHORTS PILE
PLACE SHORTS IN SHORTS PILE
END IF
ELSE IF $CLOTHES_TYPE = "nice shirt"
IF $NICE SHIRT PILE EXISTS
PLACE NICE SHIRT IN NICE SHIRT PILE
ELSE
CREATE NEW NICE SHIRT PILE
PLACE NICE SHIRT IN NICE SHIRT PILE
END IF
ELSE IF $CLOTHES_TYPE = "nice pants"
IF $NICE PANTS PILE EXISTS
PLACE NICE PANTS IN NICE PANTS PILE
ELSE
CREATE NEW NICE PANTS PILE
PLACE NICE PANTS IN NICE PANTS PILE
END IF
ELSE IF $CLOTHES_TYPE = "socks"
IF $SOCK HAS PAIR
FOLD SOCKS
IF $SOCKS PILE EXISTS
PLACE SOCKS IN SOCKS PILE
ELSE
CREATE NEW SOCKS PILE
PLACE SOCKS IN SOCKS PILE
END IF
ELSE
PLACE SINGLE SOCK IN SINGLE SOCK PILE
IF $PAIR FOUND IN SINGLE SOCK PILE
FOLD SOCKS
IF $SOCKS PILE EXISTS
PLACE SOCKS IN SOCKS PILE
ELSE
CREATE NEW SOCKS PILE
PLACE SOCKS IN SOCKS PILE
END IF
END IF
END IF
END IF
END WHILE
